CREATE TABLE letscode_genero(
	id_genero int NOT null,
	nome varchar(50) NOT null,
	PRIMARY KEY (id_genero)
);

CREATE TABLE letscode_filmes(

	id_filme int NOT null,
	nome varchar(40) NOT null,
	nota int null,
	diretor varchar(20) NOT null,
	bilheteria decimal(9,2) null,
	classificacao int null,
	id_genero int NOT null,
	FOREIGN KEY (id_genero) REFERENCES letscode_genero (id_genero), 
	PRIMARY KEY (id_filme)
	
);

CREATE TABLE letscode_ator (

	id_ator int NOT null,
	nome varchar(40) not null,
	sexo char(1),
	data_nascimento date NOT null,
	PRIMARY KEY (id_ator)
	
);

CREATE TABLE letscode_participacao (

	id_particicapao int not null,
	id_filme int not null ,
	id_ator int not null,
	PRIMARY KEY (id_particicapao),
	FOREIGN KEY (id_filme) REFERENCES letscode_filmes (id_filme),
	FOREIGN KEY (id_ator) REFERENCES letscode_ator (id_ator)
		
);
